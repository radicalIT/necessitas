\section*{Hymn}
\gresetinitiallines{1}

\begin{tabbing}
\hspace{5mm} \= \>\kill
\bf{1.}
\>Jezu, o Zbawco, coś nam wzór\\
\>Tęsknoty i miłości dał,\\
\>Od Ciebie wyszedł każdy twór,\\
\>Choć Tyś się w czasie ciałem stał.\\
\\
\bf{2.}
\>Jaki Cię łaskawości żar\\
\>Skłonił, byś zmazał zbrodni brud,\\
\>Ileż okrutną śmiercią zmarł,\\
\>Aby od śmierci zbawić lud.\\
\\
\bf{3.}
\>Zstępując na otchłani dno\\
\>Więzionych odkupiłeś sług;\\
\>Dierżysz wzycięstwa różczkę cną,\\
\>Ojcu swojemu równy Bóg.\\
\\
\bf{4.}
\>Zmusił Cię łaskawości żar,\\
\>Abyś wymazał zło i grzech,\\
\>Braciom darując grozę kar\\
\>Dajesz im widok Osób trzech.
\end{tabbing}

\clearpage
\comm{Przed Wniebowstąpieniem:}
\begin{tabbing}
    \hspace{5mm} \= \>\kill
    \bf{5.}
    \>Panie, coś stworzył każdy byt,\\
    \>Pascha gdy rozwesela skroń\\
    \>Oddalaj wszelkiej śmierci zgrzyt\\
    \>I ludu swego zawsze broń.\\
    \\
    \bf{6.}
    \>Do Ciebie Panie wznosim hymn\\
    \>Coś swoją mocą z marwych wstał\\
    \>Z Ojcem i Świętym Duchem Twym\\
    \>Byś wiekuistą chwałę miał. Amen.
\end{tabbing}\\

\comm{Od Wniebowstąpienia:}
\begin{tabbing}
    \hspace{5mm} \= \>\kill
    \bf{5.}
    \>Naszą radością sam racz być,\\
    \>Nagrodą naszą będziesz Ty,\\
    \>Chwałą nam będzie w Tobie żyć\\
    \>Po nieskończone wieków dni.\\
    \\
    \bf{6.}
    \>Do Ciebie Panie wznosim hymn,\\
    \>Coś wstąpił wzwyż nagwiezdnych ciał,\\
    \>Z Ojcem i Świętym Duchem Twym\\
    \>Byś wiekuistą chwałę miał. Amen.
\end{tabbing}
